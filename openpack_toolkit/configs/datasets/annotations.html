OPENPACK_ACTIONS_ANNOTATION = AnnotConfig(
    kind="dataset/annotation/csv",
    name="openpack-actions",
    metadata=Metadata(
        labels={
            "type": "annotation/action",
            "version": "v3.5.0",
            "dependency": "None",
            "resolution": "original",
        }
    ),
    path=DataLocation(
        dir="${path.openpack.rootdir}/${user.name}/annotation/openpack-actions",
        fname="${session}.csv",
    ),
    classes=OPENPACK_ACTIONS,
)
